クエリ1: ピアチャンネル分析

`sql

-- 同規模チャンネルのベンチマーク

SELECT

subscriber_count,

video_count,

view_count,

title,

ROUND(view_count / video_count, 0) as avg_views_per_video,

ROUND(subscriber_count / (TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), created_at, DAY) / 30), 0) as monthly_sub_growth,

TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), created_at, DAY) as channel_age_days

FROM youtube-analytics-1761171850.youtube_analytics.channels

WHERE subscriber_count BETWEEN 5000 AND 20000

ORDER BY subscriber_count DESC

LIMIT 30

`

クエリ2: バイラル動画分析

`sql

-- 10万再生以上の動画パターン

SELECT

title,

like_count,

comment_count,

duration,

ROUND(like_count * 100.0 / NULLIF(view_count, 0), 2) as like_rate,

ROUND(comment_count * 100.0 / NULLIF(view_count, 0), 2) as comment_rate,

TIMESTAMP_DIFF(CURRENT_TIMESTAMP(), published_at, DAY) as days_since_published

FROM youtube-analytics-1761171850.youtube_analytics.videos

`

クエリ3: キーワード別パフォーマンス

`sql

-- 撮影テクニック関連キーワード分析

SELECT

CASE

WHEN LOWER(title) LIKE "%カメラ%" OR LOWER(title) LIKE "%camera%" THEN "カメラ"

WHEN LOWER(title) LIKE "%撮影%" OR LOWER(title) LIKE "%撮り方%" THEN "撮影テクニック"

WHEN LOWER(title) LIKE "%leica%" OR LOWER(title) LIKE "%ライカ%" THEN "LEICA"

WHEN LOWER(title) LIKE "%iphone%" OR LOWER(title) LIKE "%スマホ%" THEN "スマホ撮影"

WHEN LOWER(title) LIKE "%写真%" OR LOWER(title) LIKE "%photo%" THEN "写真"

ELSE "その他"

END as keyword_category,

COUNT(*) as video_count,

AVG(view_count) as avg_views,

MAX(view_count) as max_views,

ROUND(AVG(like_count * 100.0 / NULLIF(view_count, 0)), 2) as avg_like_rate

FROM youtube-analytics-1761171850.youtube_analytics.videos

GROUP BY keyword_category

ORDER BY avg_views DESC

`

クエリ4: エンゲージメント分析

`sql

-- 再生回数層別エンゲージメント

SELECT

CASE

WHEN view_count < 1000 THEN "0-1K views"

WHEN view_count < 10000 THEN "1K-10K views"

WHEN view_count < 50000 THEN "10K-50K views"

WHEN view_count < 100000 THEN "50K-100K views"

ELSE "100K+ views"

END as view_tier,

COUNT(*) as video_count,

ROUND(AVG(like_count * 100.0 / NULLIF(view_count, 0)), 2) as avg_like_rate,

ROUND(AVG(comment_count * 100.0 / NULLIF(view_count, 0)), 3) as avg_comment_rate,

ROUND(AVG(view_count), 0) as avg_views

FROM youtube-analytics-1761171850.youtube_analytics.videos

GROUP BY view_tier

ORDER BY avg_views DESC

`

クエリ5: ショート vs 長尺比較

`sql

-- Shorts vs Long-form パフォーマンス

SELECT

CASE

WHEN REGEXP_CONTAINS(duration, "^PT[0-9]+S$") THEN "Shorts"

ELSE "Long-form"

END as content_type,

COUNT(*) as video_count,

ROUND(AVG(view_count), 0) as avg_views,

ROUND(AVG(like_count), 0) as avg_likes,

ROUND(AVG(like_count * 100.0 / NULLIF(view_count, 0)), 2) as avg_engagement_rate,

MAX(view_count) as top_views

FROM youtube-analytics-1761171850.youtube_analytics.videos

GROUP BY content_type

`
