
Query 2: Competitor Analysis by Subscriber Tier

```sql

SELECT

CASE

WHEN subscriber_count < 1000 THEN 'Micro (<1K)'

WHEN subscriber_count < 10000 THEN 'Small (1K-10K)'

WHEN subscriber_count < 100000 THEN 'Medium (10K-100K)'

ELSE 'Large (100K+)'

END as tier,

COUNT(*) as channel_count,

ROUND(AVG(subscriber_count)) as avg_subscribers,

ROUND(AVG(video_count)) as avg_videos,

ROUND(AVG(view_count / video_count)) as avg_views_per_video,

ROUND(AVG(view_count / subscriber_count), 2) as avg_views_per_sub

FROM `youtube-analytics-1761171850.youtube_analytics.channels`

GROUP BY tier

ORDER BY avg_subscribers;

```

Query 3: Engagement Benchmarks

```sql

SELECT

CASE

WHEN view_count < 1000 THEN '<1K views'

WHEN view_count < 10000 THEN '1K-10K views'

WHEN view_count < 100000 THEN '10K-100K views'

ELSE '100K+ views'

END as view_bucket,

COUNT(*) as video_count,

ROUND(AVG(like_count * 100.0 / NULLIF(view_count, 0)), 2) as avg_like_rate,

ROUND(AVG(comment_count * 100.0 / NULLIF(view_count, 0)), 2) as avg_comment_rate

FROM `youtube-analytics-1761171850.youtube_analytics.videos`

WHERE view_count > 0

GROUP BY view_bucket

ORDER BY avg_like_rate DESC;

```

Query 4: Keyword Performance Analysis

```sql

SELECT

keyword,

COUNT(*) as occurrences,

ROUND(AVG(view_count)) as avg_views,

ROUND(AVG(like_count * 100.0 / view_count), 2) as avg_like_rate

FROM (

SELECT

view_count,

like_count,

SPLIT(LOWER(title), ' ') as keywords

FROM `youtube-analytics-1761171850.youtube_analytics.videos`

WHERE view_count >= 50000

), UNNEST(keywords) as keyword

WHERE LENGTH(keyword) >= 2

AND (keyword LIKE '%カメラ%' OR keyword LIKE '%撮影%' OR keyword LIKE '%leica%')

GROUP BY keyword

HAVING occurrences >= 3

ORDER BY avg_views DESC

LIMIT 30;

```
