Query 2: Competitor Analysis by Subscriber Tier

`sql

SELECT

CASE

WHEN subscriber_count < 1000 THEN 'Micro (<1K)'

WHEN subscriber_count < 10000 THEN 'Small (1K-10K)'

WHEN subscriber_count < 100000 THEN 'Medium (10K-100K)'

ELSE 'Large (100K+)'

END as tier,

COUNT(*) as channel_count,

ROUND(AVG(subscriber_count)) as avg_subscribers,

ROUND(AVG(video_count)) as avg_videos,

ROUND(AVG(view_count / video_count)) as avg_views_per_video,

ROUND(AVG(view_count / subscriber_count), 2) as avg_views_per_sub

FROM youtube-analytics-1761171850.youtube_analytics.channels

GROUP BY tier

ORDER BY avg_subscribers;

`
